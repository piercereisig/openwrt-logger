#!/bin/sh /etc/rc.common

START=99
STOP=10
USE_PROCD=1

start_service() {
  procd_open_instance
  procd_set_param command /bin/sh /root/logger.sh
  procd_set_param respawn
  procd_set_param stdout /tmp/logger.out
  procd_set_param stderr /tmp/logger.err
  procd_close_instance
}

stop_service() {
  killall logger.sh 2>/dev/null
}
